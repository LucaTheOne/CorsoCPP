#include"myfuncs.h"



using namespace std;



double div(double a, double b) {

   if(b!=0) {

    return a/b;

  } else {

    throw(ErrMsg("double div(double,double)","Division by zero."));

  }

}



double log(double a) {

   if(a>0) {

    return log10(a);

  } else {

    throw(ErrMsg("double log(double)","Invalid argument."));

  }

}
